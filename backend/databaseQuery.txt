-- Automatically update `updated_at`
CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_users_updated_at
BEFORE UPDATE ON users
FOR EACH ROW
EXECUTE FUNCTION update_timestamp();

-- crime table

CREATE TYPE crime_severity AS ENUM ('minor','moderate','critical');
CREATE TYPE crime_status AS ENUM ('reported','verified','in_progress','resolved','dismissed');

CREATE TABLE crime_data (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  reporter_id uuid REFERENCES users(id) ON DELETE SET NULL,
  title TEXT NOT NULL,
  description TEXT,
  category TEXT NOT NULL,               -- e.g. 'theft', 'assault'
  severity crime_severity DEFAULT 'minor',
  status crime_status DEFAULT 'reported',
  city TEXT,
  address TEXT,
  incident_time TIMESTAMPTZ DEFAULT NOW(),
  location geography(Point,4326) NOT NULL,
  evidence_url TEXT[],                  -- 👈 store photo/video URLs
  verified_by uuid REFERENCES users(id) ON DELETE SET NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_crime_location ON crime_data USING GIST(location);
CREATE INDEX idx_crime_category ON crime_data(category);
CREATE INDEX idx_crime_city ON crime_data(city);


-- alert table
CREATE TABLE alert (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  crime_id uuid REFERENCES crime_data(id) ON DELETE SET NULL,
  title TEXT NOT NULL,
  message TEXT,
  city TEXT,
  location geography(Point,4326),
  radius_m INT DEFAULT 1000,
  alert_type TEXT DEFAULT 'general',           -- 'sos', 'warning', 'weather'
  active BOOLEAN DEFAULT TRUE,
  created_by uuid REFERENCES users(id),
  starts_at TIMESTAMPTZ DEFAULT NOW(),
  expires_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_alert_location ON alert USING GIST(location);

-- Example materialized view aggregated weekly counts (run as needed)
CREATE MATERIALIZED VIEW IF NOT EXISTS mv_crime_heat_weekly AS
SELECT ST_SnapToGrid(location::geometry, 0.002, 0.002) AS grid_geom,
       COUNT(*) AS cnt,
       MIN(ST_Centroid(ST_SnapToGrid(location::geometry, 0.002,0.002)) ) as centroid
FROM crime_data
WHERE created_at >= NOW() - INTERVAL '7 days'
GROUP BY ST_SnapToGrid(location::geometry, 0.002, 0.002);

CREATE OR REPLACE FUNCTION nearby_crimes(lat double precision, lon double precision, radius double precision)
RETURNS SETOF crime_data AS $$
BEGIN
  RETURN QUERY
  SELECT *
  FROM crime_data
  WHERE ST_DWithin(
    location,
    ST_SetSRID(ST_MakePoint(lon, lat), 4326)::geography,
    radius
  );
END;
$$ LANGUAGE plpgsql;


---
CREATE OR REPLACE FUNCTION get_nearby_crimes(
  user_lat DOUBLE PRECISION,
  user_lon DOUBLE PRECISION,
  max_distance_m INT DEFAULT 500
)
RETURNS TABLE (
  id UUID,
  title TEXT,
  category TEXT,
  severity crime_severity,
  location geography(Point,4326),
  distance_m DOUBLE PRECISION
)
AS $$
SELECT
  c.id,
  c.title,
  c.category,
  c.severity,
  c.location,
  ST_Distance(
    c.location,
    ST_SetSRID(ST_MakePoint(user_lon, user_lat), 4326)::geography
  ) AS distance_m
FROM crime_data c
WHERE ST_DWithin(
  c.location,
  ST_SetSRID(ST_MakePoint(user_lon, user_lat), 4326)::geography,
  max_distance_m
)
ORDER BY distance_m ASC;
$$ LANGUAGE sql STABLE;
